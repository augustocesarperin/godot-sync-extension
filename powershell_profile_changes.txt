# Alterações para resolver problemas com o PowerShell

# 1. Mude a configuração do PredictionViewStyle de ListView para InlineView:
Set-PSReadLineOption -PredictionViewStyle InlineView

# 2. Certifique-se que o tamanho do console está adequado:
mode con: cols=200 lines=100

# 3. Desabilite ou ajuste animações que podem causar problemas:
Set-PSReadlineOption -HistorySearchCursorMovesToEnd
Set-PSReadlineKeyHandler -Key UpArrow -Function HistorySearchBackward
Set-PSReadlineKeyHandler -Key DownArrow -Function HistorySearchForward

# 4. Otimize configurações de formatação:
$PSDefaultParameterValues['Out-File:Encoding'] = 'utf8'
$PSDefaultParameterValues['*:AutoSize'] = $true
